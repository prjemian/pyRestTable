1import pyRestTable
2t = pyRestTable.Table()
3t.labels = ('one', 'two', 'three' )
4t.rows.append( ['1,1', '1,2', '1,3',] )
5t.rows.append( ['2,1', '2,2', '2,3',] )
6t.rows.append( ['3,1', '3,2', '3,3',] )
7t.rows.append( ['4,1', '4,2', '4,3',] )
8print(t.reST(fmt='html'))
